"""
Multi-Agent Code Review Pipeline
---------------------------------
Agent 1: Code Review Agent     — syntax, style, lint, best practices
Agent 2: Security Expert Agent — vulnerabilities, OWASP, secrets, injections
Agent 3: Project Context Agent — architecture, consistency, big-picture understanding

Usage:
    python multi_agent_review.py --repo ./my-project
    python multi_agent_review.py --file ./my-project/main.py
"""

import os
import json
import argparse
from pathlib import Path
from typing import Optional
import anthropic

client = anthropic.Anthropic(api_key=os.environ.get("ANTHROPIC_API_KEY"))
MODEL = "claude-sonnet-4-20250514"

# ─────────────────────────────────────────────
# AGENT PROMPTS
# ─────────────────────────────────────────────

CODE_REVIEW_PROMPT = """
You are an expert code reviewer specializing in Python syntax, lint errors, and best practices.
Return output **only in JSON**, strictly following this schema:

{
    "agent": "code_review",
    "issues": [
        {
            "file": "<file where error occurred>",
            "line": "<line number of error>",
            "type": "<error type: syntax | lint | style | performance | best_practice>",
            "severity": "<critical | high | medium | low>",
            "original_code": "<the code that caused the error>",
            "suggested_fix": "<corrected code>",
            "explanation": [
                "<short explanation 1 in plaintext>",
                "<short explanation 2 in plaintext>",
                "<short explanation 3 in plaintext>"
            ]
        }
    ],
    "summary": "<overall code quality summary in 2-3 sentences>"
}

Rules:
- Only return valid JSON.
- Do NOT include any text outside the JSON.
- Include at least 3 explanations per issue.
- Fix the code exactly.
- Cover: unused imports, naming conventions, type hints, docstrings, complexity.
"""

SECURITY_EXPERT_PROMPT = """
You are a senior application security engineer. Analyze code for vulnerabilities.
Return output **only in JSON**, strictly following this schema:

{
    "agent": "security_expert",
    "vulnerabilities": [
        {
            "file": "<file where vulnerability found>",
            "line": "<line number>",
            "type": "<vulnerability type: injection | hardcoded_secret | insecure_deserialization | broken_auth | xss | path_traversal | other>",
            "severity": "<critical | high | medium | low | informational>",
            "owasp_category": "<OWASP Top 10 category if applicable, else null>",
            "original_code": "<vulnerable code snippet>",
            "suggested_fix": "<secure replacement code>",
            "explanation": [
                "<what the vulnerability is>",
                "<how it could be exploited>",
                "<why the fix resolves it>"
            ]
        }
    ],
    "risk_score": "<overall risk score 1-10>",
    "summary": "<security posture summary in 2-3 sentences>"
}

Rules:
- Only return valid JSON.
- Do NOT include any text outside the JSON.
- Check for: SQL injection, hardcoded credentials, insecure randomness, path traversal,
  command injection, unvalidated input, exposed secrets, insecure dependencies.
- If no issues found, return an empty vulnerabilities array with a clean summary.
"""

CONTEXT_AGENT_PROMPT = """
You are a senior software architect who deeply understands entire projects.
After reading all the provided files, return output **only in JSON**, strictly following this schema:

{
    "agent": "project_context",
    "project_overview": {
        "name": "<inferred project name>",
        "purpose": "<what this project does in 1-2 sentences>",
        "architecture": "<architecture style: monolith | microservice | library | cli | api | other>",
        "tech_stack": ["<detected technologies>"],
        "entry_points": ["<main entry files>"]
    },
    "architectural_issues": [
        {
            "file": "<file>",
            "issue": "<architectural concern>",
            "severity": "<critical | high | medium | low>",
            "recommendation": "<what to do instead>"
        }
    ],
    "consistency_issues": [
        {
            "description": "<inconsistency across files>",
            "files_affected": ["<file1>", "<file2>"],
            "recommendation": "<how to make consistent>"
        }
    ],
    "missing_components": [
        "<what this project is missing: tests, docs, error handling, logging, etc.>"
    ],
    "strengths": [
        "<things done well in this project>"
    ],
    "summary": "<holistic project assessment in 3-4 sentences>"
}

Rules:
- Only return valid JSON.
- Do NOT include any text outside the JSON.
- Think across ALL files — your job is cross-file, big-picture analysis.
- Focus on: coupling, cohesion, separation of concerns, missing abstractions, duplication.
"""

# ─────────────────────────────────────────────
# FILE LOADER
# ─────────────────────────────────────────────

SUPPORTED_EXTENSIONS = {".py", ".js", ".ts", ".jsx", ".tsx", ".go", ".java", ".rb", ".php", ".yaml", ".yml", ".toml", ".env.example"}
IGNORE_DIRS = {"node_modules", ".git", "__pycache__", ".venv", "venv", "dist", "build", ".next"}

def load_files(path: str) -> dict[str, str]:
    """Load all supported source files from a path (file or directory)."""
    files = {}
    p = Path(path)

    if p.is_file():
        files[str(p)] = p.read_text(errors="ignore")
        return files

    for root, dirs, filenames in os.walk(p):
        dirs[:] = [d for d in dirs if d not in IGNORE_DIRS]
        for fname in filenames:
            fpath = Path(root) / fname
            if fpath.suffix in SUPPORTED_EXTENSIONS:
                try:
                    files[str(fpath)] = fpath.read_text(errors="ignore")
                except Exception:
                    pass

    return files

def format_files_for_prompt(files: dict[str, str]) -> str:
    """Format loaded files into a readable block for the prompt."""
    blocks = []
    for path, content in files.items():
        blocks.append(f"### FILE: {path}\n```\n{content}\n```")
    return "\n\n".join(blocks)

# ─────────────────────────────────────────────
# AGENT RUNNER
# ─────────────────────────────────────────────

def run_agent(agent_name: str, system_prompt: str, code_context: str) -> dict:
    """Call a single agent and return parsed JSON output."""
    print(f"  ⟳  Running {agent_name}...")

    response = client.messages.create(
        model=MODEL,
        max_tokens=4096,
        system=system_prompt,
        messages=[
            {
                "role": "user",
                "content": f"Review the following code:\n\n{code_context}"
            }
        ]
    )

    raw = response.content[0].text.strip()

    # Strip markdown fences if present
    if raw.startswith("```"):
        raw = raw.split("```")[1]
        if raw.startswith("json"):
            raw = raw[4:]
        raw = raw.strip()

    try:
        return json.loads(raw)
    except json.JSONDecodeError as e:
        return {"agent": agent_name, "error": f"JSON parse failed: {e}", "raw": raw}

# ─────────────────────────────────────────────
# ORCHESTRATOR
# ─────────────────────────────────────────────

def run_pipeline(path: str, output_file: Optional[str] = None) -> dict:
    """Run all three agents and combine their results."""
    print(f"\n🔍 Loading files from: {path}")
    files = load_files(path)

    if not files:
        print("❌ No supported files found.")
        return {}

    print(f"📂 Found {len(files)} file(s): {', '.join(files.keys())}\n")
    code_context = format_files_for_prompt(files)

    print("🤖 Running multi-agent review pipeline...\n")

    results = {
        "pipeline": "multi-agent-code-review",
        "files_reviewed": list(files.keys()),
        "agents": {}
    }

    # Agent 1: Code Review
    results["agents"]["code_review"] = run_agent(
        "Code Review Agent",
        CODE_REVIEW_PROMPT,
        code_context
    )

    # Agent 2: Security Expert
    results["agents"]["security_expert"] = run_agent(
        "Security Expert Agent",
        SECURITY_EXPERT_PROMPT,
        code_context
    )

    # Agent 3: Project Context
    results["agents"]["project_context"] = run_agent(
        "Project Context Agent",
        CONTEXT_AGENT_PROMPT,
        code_context
    )

    print("\n✅ All agents complete.\n")

    # Print summaries
    for agent_key, data in results["agents"].items():
        if "error" in data:
            print(f"⚠️  {agent_key}: ERROR — {data['error']}")
        else:
            summary_key = "summary" if "summary" in data else None
            if summary_key:
                print(f"📋 [{agent_key}] {data[summary_key]}\n")

    # Save output
    out_path = output_file or "review_results.json"
    with open(out_path, "w") as f:
        json.dump(results, f, indent=2)
    print(f"💾 Full results saved to: {out_path}")

    return results

# ─────────────────────────────────────────────
# CLI
# ─────────────────────────────────────────────

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Multi-Agent Code Review Pipeline")
    group = parser.add_mutually_exclusive_group(required=True)
    group.add_argument("--repo", type=str, help="Path to a git repo or directory")
    group.add_argument("--file", type=str, help="Path to a single file")
    parser.add_argument("--output", type=str, default="review_results.json", help="Output JSON file path")

    args = parser.parse_args()
    target = args.repo or args.file

    run_pipeline(target, args.output)
